/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init(); // Permite a utilização das funções da biblioteca customizada
  GPIO_Clock_Enable(GPIOD); // Liga o sinal de clock para o barramento do Portal D

  // Configura os pinos PD0 a PD8 como Saídas Digitais para os 3 Semáforos
  for(int i = 0; i < 9; i++) {
        GPIO_Pin_Mode(GPIOD, i, OUTPUT);
    }
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while(1)
  {
      // ==============================================================
      // ESTADO 1: SEMÁFORO 1 ABERTO (Semáforos 2 e 3 no Vermelho)
      // ==============================================================
      GPIO_Write_Pin(GPIOD, PIN_0, HIGH); // Verde 1
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, LOW);
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

      // Transição: Amarelo do Semáforo 1
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, HIGH); // Amarelo 1
      GPIO_Write_Pin(GPIOD, PIN_2, LOW);
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

      // Estado de Segurança (Todos vermelhos brevemente)
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

      // ==============================================================
      // ESTADO 2: SEMÁFORO 2 ABERTO (Semáforos 1 e 3 no Vermelho)
      // ==============================================================
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, HIGH); // Verde 2
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, LOW);
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

      // Transição: Amarelo do Semáforo 2
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, HIGH); // Amarelo 2
      GPIO_Write_Pin(GPIOD, PIN_5, LOW);
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

      // Estado de Segurança (Todos vermelhos brevemente)
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

      // ==============================================================
      // ESTADO 3: SEMÁFORO 3 ABERTO (Semáforos 1 e 2 no Vermelho)
      // ==============================================================
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, HIGH); // Verde 3
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, LOW);
      Delay_ms(800);

      // Transição: Amarelo do Semáforo 3
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, HIGH); // Amarelo 3
      GPIO_Write_Pin(GPIOD, PIN_8, LOW);
      Delay_ms(800);

      // Estado de Segurança (Todos vermelhos brevemente) antes de reiniciar o loop
      GPIO_Write_Pin(GPIOD, PIN_0, LOW);
      GPIO_Write_Pin(GPIOD, PIN_1, LOW);
      GPIO_Write_Pin(GPIOD, PIN_2, HIGH); // Vermelho 1
      GPIO_Write_Pin(GPIOD, PIN_3, LOW);
      GPIO_Write_Pin(GPIOD, PIN_4, LOW);
      GPIO_Write_Pin(GPIOD, PIN_5, HIGH); // Vermelho 2
      GPIO_Write_Pin(GPIOD, PIN_6, LOW);
      GPIO_Write_Pin(GPIOD, PIN_7, LOW);
      GPIO_Write_Pin(GPIOD, PIN_8, HIGH); // Vermelho 3
      Delay_ms(800);

    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}
