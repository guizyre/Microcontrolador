/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init(); // Permite a utilização das funções da biblioteca customizada
  
  GPIO_Clock_Enable(GPIOD); // Barramento de dados (Segmentos)
  GPIO_Clock_Enable(GPIOE); // Barramento de controle (Transistores POV)

  // Configura os pinos PD0 a PD9 como Saídas Digitais (10 bits para o display)
  for(int i = 0; i < 10; i++) {
        GPIO_Pin_Mode(GPIOD, i, OUTPUT);
    }

  // Configura os pinos de chaveamento (Transistores)
  GPIO_Pin_Mode(GPIOE, PIN_2, OUTPUT); // Transistor Unidade
  GPIO_Pin_Mode(GPIOE, PIN_3, OUTPUT); // Transistor Dezena

  // Tabela de mapeamento de 10 bits para os caracteres Hexadecimais (0-F)
  uint16_t tabela_segmentos[] = {
      0b1110111111, // 0 (a,b,c,d,e,f)
      0b1110000110, // 1 (b,c)
      0b1111011011, // 2 (a,b,g,e,d)
      0b1111001111, // 3 (a,b,g,c,d)
      0b1111100110, // 4 (f,g,b,c)
      0b1111101101, // 5 (a,f,g,c,d)
      0b1111111101, // 6 (a,f,g,e,d,c)
      0b1110000111, // 7 (a,b,c)
      0b1111111111, // 8 (todos os segmentos)
      0b1111101111, // 9 (a,b,c,d,f,g)
      0b1111110111, // A (a,b,c,e,f,g)
      0b1111111100, // b (f,e,d,c,g)
      0b1110111001, // C (a,f,e,d)
      0b1111011110, // d (b,c,d,e,g)
      0b1111111001, // E (a,f,g,e,d)
      0b1111110001  // F (a,f,g,e)
  };
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while(1)
  {
        // =========================================================
        // CONTAGEM CRESCENTE (00 até FF)
        // =========================================================
        for(int i = 0; i < 16; i++) {       // Índice da Dezena
            for (int j = 0; j < 16; j++) {   // Índice da Unidade

                // Laço de Persistência da Visão (POV)
                // Repete a troca rápida 15 vezes (Aprox. 150ms visível por número)
                for(int k = 0; k < 15; k++) {
                    
                    // Passo 1: Imprime a Unidade
                    GPIO_Write_Port(GPIOD, tabela_segmentos[j]);
                    GPIO_Write_Pin(GPIOE, PIN_3, LOW);  // Corta a Dezena
                    GPIO_Write_Pin(GPIOE, PIN_2, HIGH); // Satura a Unidade
                    Delay_ms(5);

                    // Passo 2: Imprime a Dezena
                    GPIO_Write_Port(GPIOD, tabela_segmentos[i]);
                    GPIO_Write_Pin(GPIOE, PIN_2, LOW);  // Corta a Unidade
                    GPIO_Write_Pin(GPIOE, PIN_3, HIGH); // Satura a Dezena
                    Delay_ms(5);
                }
            }
        }

        // =========================================================
        // CONTAGEM DECRESCENTE (FF até 00)
        // =========================================================
        for(int i = 15; i >= 0; i--) {      // Índice da Dezena
            for (int j = 15; j >= 0; j--) {  // Índice da Unidade

                // Laço de Persistência da Visão (POV)
                for(int k = 0; k < 15; k++) {
                    
                    // Passo 1: Imprime a Unidade
                    GPIO_Write_Port(GPIOD, tabela_segmentos[j]);
                    GPIO_Write_Pin(GPIOE, PIN_3, LOW);  // Corta a Dezena
                    GPIO_Write_Pin(GPIOE, PIN_2, HIGH); // Satura a Unidade
                    Delay_ms(5);

                    // Passo 2: Imprime a Dezena
                    GPIO_Write_Port(GPIOD, tabela_segmentos[i]);
                    GPIO_Write_Pin(GPIOE, PIN_2, LOW);  // Corta a Unidade
                    GPIO_Write_Pin(GPIOE, PIN_3, HIGH); // Satura a Dezena
                    Delay_ms(5);
                }
            }
        }

      /* USER CODE END WHILE */

      /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}
