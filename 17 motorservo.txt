void EXTI4_IRQHandler() { // Botão K0 -> Move para frente (sentido 180°)
    static uint32_t ultimo = 0;
    if ((HAL_GetTick() - ultimo) > 200) { // Debounce
        if (pulso_atual + PASSO <= 2500) {
            pulso_atual += PASSO;
        } else {
            pulso_atual = 2500; // Trava no máximo
        }
        ultimo = HAL_GetTick();
    }
    EXTI_Clear_Pending(EXTI4);
}

void EXTI3_IRQHandler() { // Botão K1 -> Move para trás (sentido 0°)
    static uint32_t ultimo = 0;
    if ((HAL_GetTick() - ultimo) > 200) {
        if (pulso_atual - PASSO >= 500) {
            pulso_atual -= PASSO;
        } else {
            pulso_atual = 500; // Trava no mínimo
        }
        ultimo = HAL_GetTick();
    }
    EXTI_Clear_Pending(EXTI3);
}
