// Variáveis voláteis usadas como "Flags" pelas interrupções.
// O 'volatile' avisa ao compilador que essas variáveis podem mudar a qualquer momento (dentro da IRQ).
volatile int led1 = 0;
volatile int led2 = 0;
volatile int led3 = 0;

int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init();             // Inicializa a base de tempo para permitir o uso dos Delays
  GPIO_Clock_Enable(GPIOA);   // Habilita o clock para o barramento do Portal A
  GPIO_Clock_Enable(GPIOE);   // Habilita o clock para o barramento do Portal E


  // Configura os pinos PA6, PA7 e PA8 como Saídas Digitais (LEDs)
  GPIO_Pin_Mode(GPIOA, PIN_6, OUTPUT);
  GPIO_Pin_Mode(GPIOA, PIN_7, OUTPUT);
  GPIO_Pin_Mode(GPIOA, PIN_8, OUTPUT);
  
  // Configura os pinos PA0, PE3 e PE4 como Entradas Digitais (Botões)
  GPIO_Pin_Mode(GPIOA, PIN_0, INPUT);
  GPIO_Pin_Mode(GPIOE, PIN_4, INPUT);
  GPIO_Pin_Mode(GPIOE, PIN_3, INPUT);

  // Ativa os resistores internos:
  // PA0 como PULL-DOWN (injetará 3V3 quando apertado, ativando em borda de subida)
  GPIO_Resistor_Enable(GPIOA, PIN_0, PULL_DOWN);
  // PE4 e PA3(Atenção: verifique se não seria PE3 no seu hardware) como PULL-UP
  GPIO_Resistor_Enable(GPIOE, PIN_4, PULL_UP);
  GPIO_Resistor_Enable(GPIOA, PIN_3, PULL_UP); 


  // Garante que todos os LEDs iniciem apagados
  GPIO_Write_Pin(GPIOA, PIN_6, LOW);
  GPIO_Write_Pin(GPIOA, PIN_7, LOW);
  GPIO_Write_Pin(GPIOA, PIN_8, LOW);


  // Configura as Interrupções Externas (EXTI) para disparar na borda de subida (RISING_EDGE)
  // O evento ocorre no exato momento em que o botão muda de estado
  EXTI_Config(EXTI0, GPIOA, RISING_EDGE);
  EXTI_Config(EXTI3, GPIOE, RISING_EDGE);
  EXTI_Config(EXTI4, GPIOE, RISING_EDGE);

  // Avisa o núcleo do processador (NVIC) que ele deve aceitar e processar os alertas vindos das EXTI
  NVIC_EnableIRQ(EXTI0_IRQn);
  NVIC_EnableIRQ(EXTI4_IRQn);
  NVIC_EnableIRQ(EXTI3_IRQn);

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while(1)
  {
      // Se a interrupção do botão K_UP (EXTI0) foi acionada
      if(led1 == 1){
          // Inverte o estado do LED 1
          GPIO_Toggle_Pin(GPIOA, PIN_6);
          // Garante que os outros LEDs sejam desligados
          GPIO_Write_Pin(GPIOA, PIN_7, LOW);
          GPIO_Write_Pin(GPIOA, PIN_8, LOW);

          // Limpa todas as flags para aguardar o próximo acionamento
          led1 = 0;
          led2 = 0;
          led3 = 0;
          
          // Debounce via software
          Delay_ms(200);
      }

      // Se a interrupção do botão K0 (EXTI4) foi acionada
      if(led2 == 1){
          // Inverte o estado do LED 2
          GPIO_Toggle_Pin(GPIOA, PIN_7);
          // Garante que os outros LEDs sejam desligados
          GPIO_Write_Pin(GPIOA, PIN_6, LOW);
          GPIO_Write_Pin(GPIOA, PIN_8, LOW);
          
          // Limpa todas as flags para aguardar o próximo acionamento
          led2 = 0;
          led1 = 0;
          led3 = 0;
          
          // Debounce via software
          Delay_ms(200);
      }

      // Se a interrupção do botão K1 (EXTI3) foi acionada
      if(led3 == 1){
          // Inverte o estado do LED 3
          GPIO_Toggle_Pin(GPIOA, PIN_8);
          // Garante que os outros LEDs sejam desligados
          GPIO_Write_Pin(GPIOA, PIN_6, LOW);
          GPIO_Write_Pin(GPIOA, PIN_7, LOW);
          
          // Limpa todas as flags para aguardar o próximo acionamento
          led3 = 0;
          led2 = 0;
          led1 = 0;

          // Debounce via software
          Delay_ms(200);
      }

    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */

}
