void atividade13_03(){
	static int tempo = 0;

	GPIO_Write_Pin(GPIOA,PIN_7,HIGH);
	GPIO_Write_Pin(GPIOA,PIN_6,LOW);
	Delay_us(10000-tempo);

	GPIO_Write_Pin(GPIOA,PIN_6,HIGH);
	GPIO_Write_Pin(GPIOA,PIN_7,LOW);
	Delay_us(tempo);

	tempo += 50;

	if(tempo >= 10000){
		while(1){
			GPIO_Write_Pin(GPIOA,PIN_7,LOW);
			GPIO_Write_Pin(GPIOA,PIN_6,HIGH);
			Delay_us(tempo);

			GPIO_Write_Pin(GPIOA,PIN_7,HIGH);
			GPIO_Write_Pin(GPIOA,PIN_6,LOW);
			Delay_us(10000-tempo);
			tempo-=50;

			if(tempo <= 0) break;
		}
	}
}

void botaoK1(){
	GPIO_Write_Pin(GPIOA, PIN_6, LOW);  // D2 ON
	GPIO_Write_Pin(GPIOA, PIN_7, LOW);

	while(1){

		uint16_t leitura = 0; //Cria uma Variável de 16 bits
		//leitura = GPIOA->IDR & 1; //Atribuí o valor que está no registrador na variável leitura
		leitura = GPIO_Read_Pin(GPIOE, PIN_3);// UTILIZANDO a biblioteca faz a mesma coisa que o de cima.
		// -> Vai ler  o estado Atual do PINO3 da porta E;


		if(leitura == 0){ //Se Leitura == 0, o botão está pressionado (Pull-up)
			//Força uma Mudança de estado inicial
			GPIO_Write_Pin(GPIOA, PIN_6, HIGH);
			GPIO_Write_Pin(GPIOA, PIN_7, LOW);
			Delay_ms(200);

			while(GPIO_Read_Pin(GPIOE, PIN_3) == 0){
			//HAL_GPIO_WritePin(GPIOA, GPIO_PIN_6,GPIO_PIN_RESET); // O LED Acende (3.3V).
			GPIO_Toggle_Pin(GPIOA, PIN_6);
			//inverte os estados dos pinos
			GPIO_Toggle_Pin(GPIOA, PIN_7);
			Delay_ms(200);


			}
		}else{ // Se leitura == 1 (Botão solto)
			//Força os dois leds a apagarem
			HAL_GPIO_WritePin(GPIOA, GPIO_PIN_6,GPIO_PIN_RESET);
			HAL_GPIO_WritePin(GPIOA, GPIO_PIN_7,GPIO_PIN_RESET);
		}
	}
}
