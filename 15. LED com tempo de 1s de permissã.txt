  * * Condições de Bloqueio (Falha):
  * - Pressionar K0 antes de K-UP (Trapaça).

/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init(); // Permite a utilização das funções da biblioteca customizada
  GPIO_Clock_Enable(GPIOA);
  GPIO_Clock_Enable(GPIOE);

  // 1. Pinos dos LEDs (Saídas)
  GPIO_Pin_Mode(GPIOA, PIN_6, OUTPUT); // D2
  GPIO_Pin_Mode(GPIOA, PIN_7, OUTPUT); // D3

  // 2. Pinos dos Botões (Entradas)
  GPIO_Pin_Mode(GPIOA, PIN_0, INPUT); // K-UP (PA0)
  GPIO_Pin_Mode(GPIOE, PIN_4, INPUT); // K0 (PE4)

  // 3. Resistores (Ajustados para os botões originais da placa)
  GPIO_Resistor_Enable(GPIOA, PIN_0, PULL_DOWN); // K-UP: Garante 0V em repouso
  GPIO_Resistor_Enable(GPIOE, PIN_4, PULL_UP);   // K0: Garante 3.3V em repouso

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */

  GPIO_Write_Pin(GPIOA, PIN_6, HIGH); // D2 OFF (Lógica invertida)
  GPIO_Write_Pin(GPIOA, PIN_7, HIGH); // D3 OFF (Lógica invertida)
  
  int kup_primeiro = 0;
  uint8_t bloqueado = 0;       // Nova flag de bloqueio para estouro de tempo/trapaça
  uint32_t contador_tempo = 0; // Cronômetro de ciclos (1 ciclo = 10ms)
  uint8_t k_up_ant = 0;        // Memória do estado anterior de K-UP

  while(1)
  {
        uint8_t k_up = GPIO_Read_Pin(GPIOA, PIN_0); // K-UP (1 = pressionado)
        uint8_t k0   = GPIO_Read_Pin(GPIOE, PIN_4); // K0   (0 = pressionado)

        // ==============================================================
        // 1. DETECÇÃO DE BORDA DE SUBIDA NO K-UP
        // Checagem no momento exato do clique do botão K-UP
        // ==============================================================
        if (k_up == 1 && k_up_ant == 0) {
            if (k0 == 0) {
                // ERRO/TRAPAÇA: Se o K0 já estiver em 0 (apertado) agora, BLOQUEIA
                bloqueado = 1;
                kup_primeiro = 0;
            } else {
                // CORRETO: K0 estava em 1 (solto), então a sequência iniciou válida
                kup_primeiro = 1;
                bloqueado = 0;
                contador_tempo = 0; // Reinicia o cronômetro
            }
        }
        k_up_ant = k_up; // Salva para detectar o próximo clique na próxima iteração

        // ==============================================================
        // 2. CONTAGEM DA JANELA DE 1 SEGUNDO 
        // ==============================================================
        if (kup_primeiro == 1 && k0 == 1 && bloqueado == 0) {
            contador_tempo++;
            if (contador_tempo >= 100) { // 1 segundo (100 ciclos * 10ms)
                bloqueado = 1; // Estourou o tempo, bloqueia o acionamento
            }
        }

        // ==============================================================
        // 3. RESET TOTAL SE SOLTAR O BOTÃO K-UP
        // ==============================================================
        if (k_up == 0) {
            kup_primeiro = 0;
            bloqueado = 0;
            contador_tempo = 0;
        }

        // ==============================================================
        // 4. CONTROLE DO LED D2 (PA6)
        // Só liga se: K-UP apertado E K0 apertado E começou certo E não bloqueou
        // ==============================================================
        if (k_up == 1 && k0 == 0 && kup_primeiro == 1 && bloqueado == 0) {
            GPIO_Write_Pin(GPIOA, PIN_6, LOW);  // LIGA D2
        } else {
            GPIO_Write_Pin(GPIOA, PIN_6, HIGH); // DESLIGA D2
        }

        Delay_ms(10); // Base de tempo do cronômetro (Tick de 10ms)

      /* USER CODE END WHILE */

      /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}
