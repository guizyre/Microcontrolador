/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init();             // Inicializa base de tempo para os atrasos (Delay)
  GPIO_Clock_Enable(GPIOA);   // Habilita o clock do Portal A

  // Configura os pinos PA6 e PA7 como Saídas Digitais para os LEDs
  GPIO_Pin_Mode(GPIOA, PIN_6, OUTPUT);
  GPIO_Pin_Mode(GPIOA, PIN_7, OUTPUT);

  // Variável que atua como o "Duty Cycle" do nosso PWM por software.
  // Ela varia de 0 a 10000 microssegundos (10 ms).
  int tempo = 0;

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while(1)
  {
      // FASE 1: O brilho de um LED aumenta progressivamente enquanto o outro diminui.
      
      // Estado A: PA6 em nível BAIXO e PA7 em nível ALTO
      GPIO_Write_Pin(GPIOA, PIN_6, LOW);
      GPIO_Write_Pin(GPIOA, PIN_7, HIGH);
      Delay_us(10000 - tempo); // O tempo neste estado diminui conforme "tempo" cresce

      // Estado B: PA6 em nível ALTO e PA7 em nível BAIXO
      GPIO_Write_Pin(GPIOA, PIN_6, HIGH);
      GPIO_Write_Pin(GPIOA, PIN_7, LOW);
      Delay_us(tempo); // O tempo neste estado aumenta conforme "tempo" cresce

      // Incrementa a variável de controle, mudando a proporção de tempo (Duty Cycle)
      tempo += 50;

      // Se "tempo" atingiu o valor máximo do período (10 ms), os LEDs atingiram o limite do brilho.
      // Entramos na FASE 2 para inverter o processo de esmaecimento.
      if (tempo == 10000)
      {
          while(1)
          {
              // FASE 2: Processo reverso. Diminui de um LED, aumenta do outro.
              
              GPIO_Write_Pin(GPIOA, PIN_6, HIGH);
              GPIO_Write_Pin(GPIOA, PIN_7, LOW);
              Delay_us(tempo);

              GPIO_Write_Pin(GPIOA, PIN_6, LOW);
              GPIO_Write_Pin(GPIOA, PIN_7, HIGH);
              Delay_us(10000 - tempo);

              // Decrementa a variável de controle
              tempo -= 50;
              
              // Quando retornar a 0, quebra este loop interno para recomeçar a FASE 1
              if(tempo == 0) break;
          }
      }

    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */

}
