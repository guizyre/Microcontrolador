  * * Regra de Negócio: O LED D2 só acenderá se ambos os botões estiverem 
  * pressionados SIMULTANEAMENTE, mas exige obrigatoriamente que o botão K-UP
  * tenha sido pressionado ANTES do botão K0. 
  * * Implementação: Uma flag de estado (kup_primeiro) é armada apenas se
  * K-UP for detectado sozinho. Se o botão for solto, a flag é derrubada,

/**
  * @brief  The application entry point.
  * @retval int
  */

int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init(); // Permite a utilização das funções da biblioteca
  GPIO_Clock_Enable(GPIOA);
  GPIO_Clock_Enable(GPIOE);

  // 1. Pinos dos LEDs (Saídas)
  GPIO_Pin_Mode(GPIOA, PIN_6, OUTPUT); // D2
  GPIO_Pin_Mode(GPIOA, PIN_7, OUTPUT); // D3

  // 2. Pinos dos Botões (Entradas)
  GPIO_Pin_Mode(GPIOA, PIN_0, INPUT); // K-UP (PA0)
  GPIO_Pin_Mode(GPIOE, PIN_4, INPUT); // K0 (PE4)

  // 3. Resistores (Ajustados para os botões originais da placa)
  GPIO_Resistor_Enable(GPIOA, PIN_0, PULL_DOWN); // K-UP: Garante 0V em repouso
  GPIO_Resistor_Enable(GPIOE, PIN_4, PULL_UP);   // K0: Garante 3.3V em repouso

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */

  int kup_primeiro = 0; // Flag de estado para validar a sequência
  
  // Garante que os LEDs iniciem desligados (Lógica Invertida: HIGH = OFF)
  GPIO_Write_Pin(GPIOA, PIN_6, HIGH); // D2 OFF
  GPIO_Write_Pin(GPIOA, PIN_7, HIGH); // D3 OFF

  while(1)
  {
        uint8_t k_up = GPIO_Read_Pin(GPIOA, PIN_0); // K-UP (1 = pressionado)
        uint8_t k0   = GPIO_Read_Pin(GPIOE, PIN_4); // K0   (0 = pressionado)

        // ==============================================================
        // 1. VALIDA A SEQUÊNCIA:
        // Se K-UP está pressionado (1) e K0 ainda está solto (1),
        // significa que a ordem está correta. Habilitamos a flag.
        // ==============================================================
        if (k_up == 1 && k0 == 1) {
            kup_primeiro = 1;
        }

        // ==============================================================
        // 2. QUEBRA A SEQUÊNCIA:
        // Se o K-UP for solto (0) em qualquer momento, a sequência é quebrada.
        // A flag zera. Isso impede que apertar K0 primeiro funcione.
        // ==============================================================
        if (k_up == 0) {
            kup_primeiro = 0;
        }

        // ==============================================================
        // 3. CONTROLE DO LED:
        // Só liga se: K-UP apertado (1) E K0 apertado (0) E a sequência for válida (1)
        // ==============================================================
        if (k_up == 1 && k0 == 0 && kup_primeiro == 1) {
            GPIO_Write_Pin(GPIOA, PIN_6, LOW);  // LIGA D2 (Lógica Invertida)
        } else {
            GPIO_Write_Pin(GPIOA, PIN_6, HIGH); // DESLIGA D2 se soltar qualquer botão
        }

        Delay_ms(10); // Pequeno atraso para estabilidade e debounce de software

      /* USER CODE END WHILE */

      /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}
