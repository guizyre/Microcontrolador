/**
  * @brief  The application entry point.
  * @retval int
  */

uint8_t cont = 0; // Variável global de controle de índice da tabela

int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  
  /* USER CODE BEGIN 2 */
  Utility_Init(); // Permite a utilização das funções da biblioteca customizada
  GPIO_Clock_Enable(GPIOD); // Liga o sinal de clock para o barramento do Portal D

  // Configura os pinos PD0 a PD9 como Saídas Digitais para o Display
  for(int i = 0; i < 10; i++) {
        GPIO_Pin_Mode(GPIOD, i, OUTPUT);
    }

  // Tabela de mapeamento de 10 bits para os caracteres Hexadecimais (0-F)
  uint16_t tabela_segmentos[] = {
      0b1110111111, // 0 (a,b,c,d,e,f)
      0b1110000110, // 1 (b,c)
      0b1111011011, // 2 (a,b,g,e,d)
      0b1111001111, // 3 (a,b,g,c,d)
      0b1111100110, // 4 (f,g,b,c)
      0b1111101101, // 5 (a,f,g,c,d)
      0b1111111101, // 6 (a,f,g,e,d,c)
      0b1110000111, // 7 (a,b,c)
      0b1111111111, // 8 (todos os segmentos)
      0b1111101111, // 9 (a,b,c,d,f,g)
      0b1111110111, // A (a,b,c,e,f,g)
      0b1111111100, // b (f,e,d,c,g)
      0b1110111001, // C (a,f,e,d)
      0b1111011110, // d (b,c,d,e,g)
      0b1111111001, // E (a,f,g,e,d)
      0b1111110001  // F (a,f,g,e)
  };

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  cont = 0; // Garante inicialização limpa antes do laço
  
  while(1)
  {
      // Contagem Crescente: 0 ao E (15 iterações)
      // Para no índice 15 para o próximo laço iniciar do 'F'
      for(int i = 0; i < 15; i++){
          GPIO_Write_Port(GPIOD, tabela_segmentos[cont]);
          cont++;
          Delay_ms(500);
      }
      
      // Contagem Decrescente: F ao 1 (15 iterações)
      // Para no índice 0 para o laço superior reiniciar do '0'
      for(int i = 0; i < 15; i++){
          GPIO_Write_Port(GPIOD, tabela_segmentos[cont]);
          cont--;
          Delay_ms(500);
      }

    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}
